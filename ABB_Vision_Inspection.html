! MODULE: SortingSystem · ABB IRB 1200 · RAPID v6
MODULE SortingSystem

  PERS tooldata  tGripper := [TRUE,[[0,0,120],[1,0,0,0]],...];
  CONST robtarget pBin_RED   := [[300,-150,80],...];
  CONST robtarget pBin_BLUE  := [[300,0,80],...];
  CONST robtarget pBin_GREEN := [[300,150,80],...];
  CONST robtarget pHome      := [[0,0,400],...];
  VAR   num       nColor     := 0;
  VAR   robtarget pPick;

PROC main()
  MoveJ pHome, v400, z50, tGripper;
  WHILE TRUE DO
    WaitDI di_RobStart, 1;        ! Wait PLC start
    SetDO  do_RobBusy, 1;
    nColor := GInput(giColorCode);
    pPick  := ReadVisionPos();
    MoveJ Offs(pPick,0,0,100), v600, z10, tGripper;
    MoveL pPick, v80, fine, tGripper;
    SetDO do_GripClose, 1;
    WaitTime 0.3;
    MoveL Offs(pPick,0,0,150), v400, z20, tGripper;
    SortToBin nColor;
    SetDO do_RobBusy, 0;
    SetDO do_RobDone, 1;
    WaitTime 0.1;
    SetDO do_RobDone, 0;
  ENDWHILE
ENDPROC

PROC SortToBin(num id)
  VAR robtarget pT;
  TEST id
    CASE 1: pT:=pBin_RED;
    CASE 2: pT:=pBin_BLUE;
    CASE 3: pT:=pBin_GREEN;
    DEFAULT: ErrLog "Unknown color!";
  ENDTEST
  MoveJ Offs(pT,0,0,120), v600, z10, tGripper;
  MoveL pT, v100, fine, tGripper;
  SetDO do_GripOpen, 1;
  WaitTime 0.2;
  MoveL Offs(pT,0,0,120), v400, z20, tGripper;
  MoveJ pHome, v600, z50, tGripper;
ENDPROC

ENDMODULE
